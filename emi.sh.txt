
#!/bin/bash


#if the fomulae is this  
#$P × $r × (1 + r)$n/((1 + r)$n - 1

echo "Enter the amount of loan RS :"
read p
echo "Enter the rate of interest in %:"
read r
echo "Enter the loan tenure in months :"
read n


rate=`expr ($r/(12 \*100))`
echo $rate
var1=`expr 1 + $rate`
echo $var1
var2=`expr $var1 * $n`
echo $var2
var3=`expr $p \* $rate \* $var2`
echo $var3
emi=`expr $var3 / $var2 - 1
echo "EMI : rs " $emi
